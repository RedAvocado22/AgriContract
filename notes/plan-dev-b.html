1. Static factory offer()

public static Contract offer(
    String contractId, String listingId,
    String sellerId, String buyerId,
    String productName, Quantity quantity,
    Money agreedPrice, LocalDate deliveryDeadline
) {
    // validate: không được null, sellerId != buyerId
    Contract c = new Contract();
    c.contractId = new ContractId(contractId);
    c.listingId = listingId;
    c.sellerId = sellerId;
    c.buyerId = buyerId;
    c.productName = productName;
    c.quantity = quantity;
    c.agreedPrice = agreedPrice;
    c.deliveryDeadline = deliveryDeadline;
    c.status = ContractStatus.OFFERED;
    c.signatories = new HashSet<>();
    c.domainEvents = new ArrayList<>();
    c.createdAt = c.updatedAt = Instant.now();
    c.domainEvents.add(ContractOfferedEvent.of(contractId, sellerId, buyerId));
    return c;
}

2. sign(userId) — Logic phức tạp nhất

public void sign(String userId) {
    // 1. Validate caller là seller hoặc buyer của contract này
    if (!userId.equals(this.sellerId) && !userId.equals(this.buyerId)) {
        throw new IllegalArgumentException(
            "User " + userId + " is not a party to this contract");
    }

    // 2. Contract phải ở OFFERED để sign
    if (this.status != ContractStatus.OFFERED) {
        throw new IllegalStateException(
            "Cannot sign contract with status: " + this.status);
    }

    // 3. Thêm vào signatories set
    this.signatories.add(userId);
    this.updatedAt = Instant.now();

    // 4. Chỉ SIGNED khi đủ 2 người (cả seller và buyer đều đã ký)
    if (this.signatories.contains(this.sellerId)
        && this.signatories.contains(this.buyerId)) {

        this.status = ContractStatus.SIGNED;
        this.domainEvents.add(ContractSignedEvent.of(
            this.contractId.value(), this.listingId,
            this.sellerId, this.buyerId,
            this.agreedPrice, this.deliveryDeadline
        ));
    }
    // Nếu mới 1 người ký → không emit event, giữ OFFERED
}
💡
Ai ký trước cũng được Seller hay buyer ký trước đều không quan trọng. Contract chỉ chuyển sang SIGNED khi cả 2 đã ký. signatories là Set<String> nên không bị duplicate.

3. activate()

public void activate() {
    if (this.status != ContractStatus.SIGNED)
        throw new IllegalStateException("Cannot activate contract with status: " + this.status);
    this.status = ContractStatus.ACTIVE;
    this.updatedAt = Instant.now();
}

4. Cancel flows

public void cancel(String userId, String reason) {
    validateCancellable();
    this.cancelledBy = userId.equals(this.buyerId) ? "BUYER" : "SELLER";
    this.status = ContractStatus.CANCELLED;
    this.cancelReason = reason;
    this.updatedAt = Instant.now();
    this.domainEvents.add(ContractCancelledEvent.of(
        this.contractId.value(), this.cancelledBy, reason,
        this.sellerId, this.buyerId));
}

private void validateCancellable() {
    if (this.status != ContractStatus.OFFERED
        && this.status != ContractStatus.SIGNED
        && this.status != ContractStatus.ACTIVE) {
        throw new IllegalStateException(
            "Cannot cancel contract with status: " + this.status);
    }
}

4. confirmDelivery(), settle(), dispute(), arbitrate()

public void confirmDelivery(String buyerId) {
    if (!buyerId.equals(this.buyerId))
        throw new IllegalArgumentException("Only buyer can confirm delivery");
    if (this.status != ContractStatus.ACTIVE)
        throw new IllegalStateException("Cannot confirm delivery: status is " + this.status);

    this.status = ContractStatus.DELIVERED;
    this.updatedAt = Instant.now();
    this.domainEvents.add(GoodsDeliveredEvent.of(this.contractId.value(),
        this.sellerId, this.buyerId));
}

public void settle() {
    // Gọi từ EscrowReleasedEventHandler — không validate actor
    if (this.status != ContractStatus.DELIVERED)
        throw new IllegalStateException(...);
    this.status = ContractStatus.SETTLED;
    this.updatedAt = Instant.now();
    this.domainEvents.add(ContractSettledEvent.of(this.contractId.value()));
}

public void dispute(String buyerId) {
    if (!buyerId.equals(this.buyerId))
        throw new IllegalArgumentException("Only buyer can open dispute");
    if (this.status != ContractStatus.DELIVERED)
        throw new IllegalStateException(...);
    this.status = ContractStatus.DISPUTED;
    this.updatedAt = Instant.now();
    this.domainEvents.add(ContractDisputedEvent.of(this.contractId.value()));
}

public void arbitrate() {
    // Gọi từ EscrowArbitratedEventHandler
    if (this.status != ContractStatus.DISPUTED)
        throw new IllegalStateException(...);
    this.status = ContractStatus.ARBITRATED;
    this.updatedAt = Instant.now();
    this.domainEvents.add(ContractArbitratedEvent.of(this.contractId.value()));
}

5. pullDomainEvents()

public List<DomainEvent> pullDomainEvents() {
    List<DomainEvent> events = List.copyOf(this.domainEvents);
    this.domainEvents.clear();   // drain — không emit lại
    return events;
}

// Helper methods trong mapper (dùng Jackson ObjectMapper):
default Set<String> parseSignatories(String json) {
    if (json == null || json.isBlank()) return new HashSet<>();
    return objectMapper.readValue(json, new TypeReference<Set<String>>(){});
}
default String serializeSignatories(Set<String> set) {
    return objectMapper.writeValueAsString(set);  // → ["userId1","userId2"]
}

⚡ Transaction Boundary — Critical

Mỗi use case phải save contract + domain events trong cùng 1 @Transactional. Nếu tách ra 2 transaction → sẽ có state inconsistency khi service crash giữa chừng.

@Transactional
public ContractResponse execute(Command cmd) {
    // 1. business logic
    contract.sign(userId);

    // 2. save contract
    contractRepository.save(contract);

    // 3. save outbox events — CÙNG TRANSACTION
    for (DomainEvent e : contract.pullDomainEvents()) {
        domainEventRepository.save(e);
    }

    return toResponse(contract);
    // TX commit → cả contract update + outbox insert được commit cùng lúc
}

CreateContractUseCase

@Transactional
public ContractResponse execute(CreateContractCommand cmd) {
    // 1. Lookup listing từ product-service qua Feign
    ListingResponse listing = productServiceClient.getListing(cmd.listingId());
    if (!"ACTIVE".equals(listing.status())) {
        throw new IllegalStateException("Listing is not ACTIVE");
    }

    // 2. Tạo contract với data snapshot từ listing
    String contractId = UUID.randomUUID().toString();
    Contract contract = Contract.offer(
        contractId,
        listing.listingId(),
        listing.sellerId(),   // sellerId từ listing
        cmd.buyerId(),        // buyerId từ JWT (inject qua command)
        listing.productName(), // snapshot
        new Quantity(listing.quantity(), listing.quantityUnit()),
        new Money(listing.priceFloor(), listing.currency()),
        listing.deliveryDeadline()
    );

    // 3. Save contract + outbox events trong cùng TX
    contractRepository.save(contract);
    contract.pullDomainEvents().forEach(domainEventRepository::save);

    return toResponse(contract);
}

@Transactional
public ContractResponse execute(String contractId, String userId) {
    Contract contract = contractRepository.findById(contractId)
        .orElseThrow(() -> new NotFoundException("Contract not found"));

    contract.sign(userId);

    // Nếu contract vừa SIGNED → gọi Feign close listing TRƯỚC khi save

Outbox Poller

contract-service/.../infrastructure/messaging/OutboxPoller.java

Scheduler chạy mỗi 1 giây, đọc PENDING events từ DB và publish lên RabbitMQ:

@Component
public class OutboxPoller {

    @Scheduled(fixedDelay = 1000)
    @Transactional
    public void poll() {
        List<ContractDomainEventJpaEntity> pending =
            domainEventJpaRepo.findTop50ByStatusOrderByCreatedAtAsc("PENDING");

        for (ContractDomainEventJpaEntity event : pending) {
            try {
                String routingKey = toRoutingKey(event.getEventType());
                rabbitTemplate.convertAndSend(
                    "agricontract.contracts",  // exchange
                    routingKey,
                    event.getPayload()          // JSON string
                );
                event.setStatus("PUBLISHED");
                event.setPublishedAt(Instant.now());

            } catch (Exception e) {
                log.error("Failed to publish event {}: {}", event.getEventId(), e.getMessage());
                event.setRetryCount(event.getRetryCount() + 1);
                if (event.getRetryCount() >= 3) {
                    event.setStatus("FAILED");
                }
            }
            domainEventJpaRepo.save(event);
        }
    }

    private String toRoutingKey(String eventType) {
        return switch (eventType) {
            case "ContractOfferedEvent"    -> "contract.offered";
            case "ContractSignedEvent"     -> "contract.signed";
            case "GoodsDeliveredEvent"     -> "contract.delivered";
            case "ContractCancelledEvent"  -> "contract.cancelled";
            case "ContractDisputedEvent"   -> "contract.disputed";
            case "ContractSettledEvent"    -> "contract.settled";
            case "ContractArbitratedEvent" -> "contract.arbitrated";
            default -> throw new IllegalArgumentException("Unknown event: " + eventType);
        };
    }
}

Escrow Event Consumer

contract-service cần nhận 2 events từ escrow-service để complete async flow:

contract-service/.../infrastructure/messaging/EscrowEventConsumer.java 
@Component
public class EscrowEventConsumer {

    @RabbitListener(queues = "contract-svc.escrow.locked")
    @Transactional
    public void onEscrowLocked(String payload) {
        Map<String, Object> msg = parseJson(payload);
        String contractId = (String) msg.get("contractId");
        String eventId    = (String) msg.get("eventId");
        if (processedEventRepository.existsByEventId(eventId)) return;
        Contract contract = contractRepository.findById(contractId)
            .orElseThrow(() -> new NotFoundException("Contract not found: " + contractId));
        contract.activate();
        contractRepository.save(contract);
        contract.pullDomainEvents().forEach(domainEventRepository::save);
        processedEventRepository.save(new ProcessedEvent(eventId));
    }

    // escrow.released → settle contract
    @RabbitListener(queues = "contract-service.escrow.released")
    @Transactional
    public void onEscrowReleased(String payload) {
        Map<String, Object> msg = parseJson(payload);
        String contractId = (String) msg.get("contractId");

        Contract contract = contractRepository.findById(contractId)
            .orElseThrow(() -> new NotFoundException("Contract not found: " + contractId));

        contract.settle();
        contractRepository.save(contract);
        contract.pullDomainEvents().forEach(domainEventRepository::save);
    }
}
💡
Queue names phải match với Dev C Queue contract-service.escrow.locked và contract-service.escrow.released phải được khai báo trong RabbitMQConfig của contract-service và bind đúng vào exchange agricontract.escrow của escrow-service. Coordinate với Dev C về tên queue.

EscrowAccount Aggregate

EscrowAccount giữ tiền escrow và duy trì append-only ledger — tất cả EscrowTransaction chỉ INSERT, không bao giờ UPDATE hay DELETE.

escrow-service/.../domain/model/EscrowAccount.java

lock() — tạo EscrowAccount

public static EscrowAccount lock(
    String escrowId, String contractId,
    String buyerUserId, String sellerUserId,
    BigDecimal totalAmount, String currency
) {
    EscrowAccount a = new EscrowAccount();
    a.escrowId     = escrowId;
    a.contractId   = contractId;   // UNIQUE — idempotency key
    a.buyerUserId  = buyerUserId;
    a.sellerUserId = sellerUserId;
    a.totalAmount  = totalAmount;
    a.currency     = currency;
    a.status       = EscrowStatus.LOCKED;
    a.transactions = new ArrayList<>();
    a.createdAt    = a.updatedAt = Instant.now();
    a.transactions.add(new EscrowTransaction(
        UUID.randomUUID().toString(), TransactionType.LOCK, totalAmount, "escrow locked"));
    return a;
}

release()

public void release() {
    if (this.status != EscrowStatus.LOCKED)
        throw new IllegalStateException("Cannot release escrow with status: " + this.status);
    this.status = EscrowStatus.RELEASED;
    this.updatedAt = Instant.now();
    this.transactions.add(new EscrowTransaction(
        UUID.randomUUID().toString(), TransactionType.RELEASE, totalAmount, "released to seller"));
}

penalizeBuyer() + penalizeSeller()

public void penalizeBuyer(BigDecimal penaltyRate) {
    if (this.status != EscrowStatus.LOCKED)
        throw new IllegalStateException(...);
    BigDecimal penalty = totalAmount.multiply(penaltyRate);     // e.g. 0.10
    BigDecimal refund  = totalAmount.subtract(penalty);
    this.status = EscrowStatus.PENALIZED_BUYER;
    this.updatedAt = Instant.now();
    // Append-only: 2 ledger entries
    this.transactions.add(new EscrowTransaction(..., TransactionType.PENALIZE_BUYER, penalty,  "penalty to seller"));
    this.transactions.add(new EscrowTransaction(..., TransactionType.RELEASE, refund, "refund to buyer"));
}

public void penalizeSeller() {
    if (this.status != EscrowStatus.LOCKED) throw new IllegalStateException(...);
    this.status = EscrowStatus.PENALIZED_SELLER;
    this.updatedAt = Instant.now();
    // Phase 1: full refund to buyer, seller penalty chỉ log
    this.transactions.add(new EscrowTransaction(..., TransactionType.PENALIZE_SELLER, BigDecimal.ZERO, "penalty logged"));
    this.transactions.add(new EscrowTransaction(..., TransactionType.RELEASE, totalAmount, "full refund to buyer"));
}

arbitrate()

public void arbitrate(BigDecimal buyerAmount, BigDecimal sellerAmount) {
    if (this.status != EscrowStatus.LOCKED && this.status != EscrowStatus.DISPUTED)
        throw new IllegalStateException(...);
    if (buyerAmount.add(sellerAmount).compareTo(totalAmount) != 0)
        throw new IllegalArgumentException("buyerAmount + sellerAmount must equal totalAmount");
    this.status = EscrowStatus.ARBITRATED;
    this.updatedAt = Instant.now();
    this.transactions.add(new EscrowTransaction(..., TransactionType.ARBITRATION, sellerAmount, "admin decision"));
}

@Component
public class ContractEventConsumer {

    @RabbitListener(queues = "escrow-svc.contract.signed")
    @Transactional
    public void onContractSigned(String payload) {
        Map<String, Object> msg = parseJson(payload);
        String contractId = (String) msg.get("contractId");

        // Idempotency: skip nếu escrow đã tồn tại cho contractId này
        if (escrowAccountJpaRepo.existsByContractId(contractId)) {
            log.warn("Duplicate ContractSignedEvent for contractId {}, skipping", contractId);
            return;
        }

        EscrowAccount account = EscrowAccount.lock(
            UUID.randomUUID().toString(),
            contractId,
            (String) msg.get("buyerId"),
            (String) msg.get("sellerId"),
            new BigDecimal(msg.get("agreedPrice").toString()),
            (String) msg.get("currency")
        );
        escrowAccountRepository.save(account);

        // Publish escrow.locked → contract-service sẽ nhận
        rabbitTemplate.convertAndSend("agricontract.escrow", "escrow.locked",
            objectMapper.writeValueAsString(Map.of("contractId", contractId)));
    }

    @RabbitListener(queues = "escrow-svc.contract.delivered")
    @Transactional
    public void onContractDelivered(String payload) {
        String contractId = extractContractId(payload);
        EscrowAccount account = findByContractId(contractId);
        account.release();
        escrowAccountRepository.save(account);

        rabbitTemplate.convertAndSend("agricontract.escrow", "escrow.released",
            objectMapper.writeValueAsString(Map.of("contractId", contractId)));
    }

    @RabbitListener(queues = "escrow-svc.contract.cancelled")
    @Transactional
    public void onContractCancelled(String payload) {
        Map<String, Object> msg = parseJson(payload);
        String contractId  = (String) msg.get("contractId");
        String cancelledBy = (String) msg.get("cancelledBy");

        EscrowAccount account = findByContractId(contractId);
        if ("BUYER".equals(cancelledBy)) {
            account.penalizeBuyer(new BigDecimal("0.10"));  // 10% penalty
        } else {
            account.penalizeSeller();
        }
        escrowAccountRepository.save(account);

        Map<String, Object> event = Map.of(
            "contractId", contractId,
            "penalizedParty", cancelledBy,
            "penaltyAmount", account.getPenaltyAmount()
        );
        rabbitTemplate.convertAndSend("agricontract.escrow", "escrow.penalized",
            objectMapper.writeValueAsString(event));
    }
}

@Configuration
public class RabbitMQConfig {

    // Exchange escrow-service publish vào
    @Bean public TopicExchange escrowExchange() {
        return new TopicExchange("agricontract.escrow");
    }

    // Exchange escrow-service subscribe từ
    @Bean public TopicExchange contractsExchange() {
        return new TopicExchange("agricontract.contracts");
    }

    @Bean public Queue signedQueue()    { return QueueBuilder.durable("escrow-svc.contract.signed").build(); }
    @Bean public Queue deliveredQueue() { return QueueBuilder.durable("escrow-svc.contract.delivered").build(); }
    @Bean public Queue cancelledQueue() { return QueueBuilder.durable("escrow-svc.contract.cancelled").build(); }

    @Bean public Binding bindSigned()    { return BindingBuilder.bind(signedQueue()).to(contractsExchange()).with("contract.signed"); }
    @Bean public Binding bindDelivered() { return BindingBuilder.bind(deliveredQueue()).to(contractsExchange()).with("contract.delivered"); }
    @Bean public Binding bindCancelled() { return BindingBuilder.bind(cancelledQueue()).to(contractsExchange()).with("contract.cancelled"); }
}
