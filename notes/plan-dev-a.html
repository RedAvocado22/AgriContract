public void verify() {
    if (this.verificationStatus != VerificationStatus.PENDING)
        throw new IllegalStateException("Cannot verify from status: " + verificationStatus);
    this.verificationStatus = VerificationStatus.VERIFIED;
    this.updatedAt = Instant.now();
}

public void reject() {
    if (this.verificationStatus != VerificationStatus.PENDING)
        throw new IllegalStateException(...);
    this.verificationStatus = VerificationStatus.REJECTED;
    this.updatedAt = Instant.now();
}

public void updateContactInfo(String phone, String address) {
    this.phone = phone; this.address = address;
    this.updatedAt = Instant.now();
}

        if (userProfileRepository.existsById(cmd.userId())) {
            return toResponse(userProfileRepository.findById(cmd.userId()).get());
        }
        UserProfile profile = UserProfile.create(

            cmd.email(), cmd.phone(), cmd.address()
        );
        return toResponse(userProfileRepository.save(profile));
    }
}

        throw new IllegalStateException("Cannot close, status: " + this.status);
    this.status = ListingStatus.CLOSED;
    this.updatedAt = Instant.now();
}

public void expire() {
    if (this.status != ListingStatus.ACTIVE) throw new IllegalStateException(...);
    this.status = ListingStatus.EXPIRED;
    this.updatedAt = Instant.now();
}

        return new TopicExchange("agricontract.escrow");
    }

    }
    @Bean public Queue notifEscrowQueue() {
        return QueueBuilder.durable("notif-svc.escrow").build();

    // Wildcard # — nhận TẤT CẢ events từ cả 2 exchanges

        return BindingBuilder.bind(notifContractsQueue()).to(contractsExchange()).with("#");
    }
    @Bean public Binding bindNotifEscrow() {
        return BindingBuilder.bind(notifEscrowQueue()).to(escrowExchange()).with("#");
    }

    @Bean public Jackson2JsonMessageConverter messageConverter() {
        return new Jackson2JsonMessageConverter();
    }
}

Pattern xử lý — bắt buộc theo đúng thứ tự

@RabbitListener(queues = "notif-svc.contracts")
public void onContractEvent(String payload) {
    Map<String, Object> msg = parseJson(payload);
    String eventId   = (String) msg.get("eventId");
    String eventType = (String) msg.get("eventType");

    // 1. Idempotency — bỏ qua duplicate (RabbitMQ at-least-once delivery)
    if (notifRepo.existsByEventId(eventId)) {
        log.debug("Event {} already processed, skipping", eventId);
        return;
    }

    // 2. Persist PENDING trước khi gửi
    NotificationLog log = buildLog(eventId, eventType, msg);
    notifRepo.save(log);

    // 3. Dispatch theo event type
    switch (eventType) {
        case "ContractSignedEvent"    -> handleContractSigned(msg, log);
        case "GoodsDeliveredEvent"     -> handleGoodsDelivered(msg, log);
        case "ContractCancelledEvent"  -> handleContractCancelled(msg, log);
        case "ContractDisputedEvent"   -> handleContractDisputed(msg, log);
        case "ContractSettledEvent"    -> handleContractSettled(msg, log);
    }
}

@RabbitListener(queues = "notif-svc.escrow")
public void onEscrowEvent(String payload) {
    Map<String, Object> msg = parseJson(payload);
    String eventId   = (String) msg.get("eventId");
    String eventType = (String) msg.get("eventType");

    if (notifRepo.existsByEventId(eventId)) return;

    switch (eventType) {
        case "EscrowLockedEvent"    -> handleEscrowLocked(msg);
        case "EscrowReleasedEvent"  -> handleEscrowReleased(msg);
        case "EscrowPenalizedEvent" -> handleEscrowPenalized(msg);
    }
}

Retry logic — 3 lần, backoff 1s / 2s / 4s

private void sendEmailWithRetry(String to, String subject, String body) {
    long[] backoff = {1000L, 2000L, 4000L};
    for (int i = 0; i < 3; i++) {
        try {
            SimpleMailMessage mail = new SimpleMailMessage();
            mail.setTo(to); mail.setSubject(subject); mail.setText(body);
            mailSender.send(mail);
            return;
        } catch (MailException e) {
            if (i < 2) Thread.sleep(backoff[i]);
        }
    }
    // Sau 3 lần: update log.status = FAILED
}
