public static UserProfile create(
    String userId,          // = Keycloak sub claim (UUID string)
    String organizationName,
    Role role,

2. verify() và reject()

public void verify() {
    if (this.verificationStatus != VerificationStatus.PENDING) {
        throw new IllegalStateException(
            "Cannot verify: current status is " + this.verificationStatus);
    }
    this.verificationStatus = VerificationStatus.VERIFIED;
    this.updatedAt = Instant.now();
}

public void reject() {
    if (this.verificationStatus != VerificationStatus.PENDING) {
        throw new IllegalStateException(...);
    }
    this.verificationStatus = VerificationStatus.REJECTED;
    this.updatedAt = Instant.now();
}

public void updateContactInfo(String email, String phone, String address) {
    this.contactInfo = new ContactInfo(email, phone, address);
    this.updatedAt = Instant.now();
}

Implement UserProfileRepositoryImpl

user-service/.../infrastructure/persistence/UserProfileRepositoryImpl.java 
@Repository
public class UserProfileRepositoryImpl implements UserProfileRepository {

    private final UserProfileJpaRepository jpaRepo;
    private final UserProfileMapper mapper;

    public UserProfile save(UserProfile domain) {
        UserProfileJpaEntity entity = mapper.toEntity(domain);
        UserProfileJpaEntity saved = jpaRepo.save(entity);
        return mapper.toDomain(saved);
    }

    public Optional<UserProfile> findById(String userId) {
        return jpaRepo.findByUserId(userId).map(mapper::toDomain);
    }

    public boolean existsById(String userId) {
        return jpaRepo.existsByUserId(userId);
    }
}

public static Listing create(
    String listingId, String sellerId,
    String productId, String productName,  // productName là SNAPSHOT
    Quantity quantity, Money priceFloor,
    LocalDate deliveryDeadline
) {
    Listing l = new Listing();
    l.listingId = new ListingId(listingId);
    l.sellerId = sellerId;
    l.productId = productId;
    l.productName = productName; // copy tại thời điểm tạo, không dùng reference
    l.quantity = quantity;
    l.priceFloor = priceFloor;
    l.deliveryDeadline = deliveryDeadline;
    l.status = ListingStatus.ACTIVE;
    l.createdAt = l.updatedAt = Instant.now();
    return l;
}

public void close() {
    if (this.status != ListingStatus.ACTIVE) {
        throw new IllegalStateException(
            "Cannot close listing with status: " + this.status);
    }
    this.status = ListingStatus.CLOSED;
    this.updatedAt = Instant.now();
}

public void expire() {
    if (this.status != ListingStatus.ACTIVE) {
        throw new IllegalStateException(...);
    }
    this.status = ListingStatus.EXPIRED;
    this.updatedAt = Instant.now();
}
💡
Tại sao snapshot productName? Khi seller sửa Product sau này, các Listing đã tạo không bị ảnh hưởng. Contract-service sẽ lấy productName từ Listing tại thời điểm tạo contract — đây là invariant nghiệp vụ quan trọng.

@Service
public class RegisterUserUseCaseImpl implements RegisterUserUseCase {

    @Transactional
    public UserProfileResponse execute(RegisterUserCommand cmd) {
        if (repository.existsById(cmd.userId())) {
            throw new IllegalStateException("User already registered: " + cmd.userId());
        }
        UserProfile profile = UserProfile.create(
            cmd.userId(), cmd.organizationName(), cmd.role(),
            cmd.email(), cmd.phone(), cmd.address()
        );
        UserProfile saved = repository.save(profile);
        return toResponse(saved);
    }
}

public class UserController {

    @PostMapping("/register")
    public ResponseEntity<UserProfileResponse> register(
        @RequestBody @Valid RegisterUserRequest req,
        @AuthenticationPrincipal Jwt jwt   // inject từ Spring Security
    ) {
        String userId = jwt.getSubject();   // Keycloak sub
        String email  = jwt.getClaimAsString("email");
        RegisterUserCommand cmd = new RegisterUserCommand(
            userId, req.organizationName(), req.role(), email, req.phone(), req.address()
        );
        return ResponseEntity.status(201).body(registerUserUseCase.execute(cmd));
    }

        return getUserProfileUseCase.execute(jwt.getSubject());

        return getUserProfileUseCase.execute(userId);
    }
}

@Transactional
public ListingResponse execute(CreateListingCommand cmd) {
    // 1. Lookup product để lấy productName snapshot
    Product product = productRepository.findById(cmd.productId())
        .orElseThrow(() -> new NotFoundException("Product not found"));

    // 2. Tạo listing với productName snapshot
    String listingId = UUID.randomUUID().toString();
    Listing listing = Listing.create(
        listingId, cmd.sellerId(),
        product.getProductId().value(),
        product.getName(),      // ← snapshot tại đây
        new Quantity(cmd.quantity(), cmd.quantityUnit()),
        new Money(cmd.priceFloor(), cmd.currency()),
        cmd.deliveryDeadline()
    );
    return toResponse(listingRepository.save(listing));
}

authStore.ts (Zustand)

frontend/src/store/authStore.ts 
import { create } from 'zustand';

interface AuthState {
  token:  string | null;
  userId: string | null;
  role:   'SELLER' | 'BUYER' | 'ADMIN' | null;
  setAuth: (token: string, userId: string, role: string) => void;
  clearAuth: () => void;
}

export const useAuthStore = create<AuthState>((set) => ({
  token: null, userId: null, role: null,
  setAuth: (token, userId, role) => set({ token, userId, role: role as any }),
  clearAuth: () => set({ token: null, userId: null, role: null }),
}));
🔒
Cấm lưu token vào localStorage Token chỉ được lưu trong Zustand (memory). Khi refresh trang → Keycloak JS tự silent-refresh. Nếu vứt vào localStorage thì dính XSS attack dễ dàng lấy token.

Axios client.ts

frontend/src/api/client.ts 
import axios from 'axios';
import { useAuthStore } from '../store/authStore';
import keycloak from '../auth/keycloak';

const apiClient = axios.create({
  baseURL: 'http://localhost:8888',  // API Gateway port 8888
});

// Request interceptor: inject token
apiClient.interceptors.request.use((config) => {
  const token = useAuthStore.getState().token;
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

// Response interceptor: handle 401 + 5xx
apiClient.interceptors.response.use(
  (res) => res,
  async (error) => {
    if (error.response?.status === 401) {
      try {
        await keycloak.updateToken(30);   // refresh token
        useAuthStore.getState().setAuth(keycloak.token!, ...);
        return apiClient(error.config);   // retry request
      } catch {
        keycloak.logout();
      }
    }
    if (error.response?.status >= 500) {
      // show toast "Lỗi server, thử lại sau"
    }
    return Promise.reject(error);
  }
);

export default apiClient;
