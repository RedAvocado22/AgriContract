  listing_id        VARCHAR(36)   PRIMARY KEY,
  product_id        VARCHAR(36)   NOT NULL,

  contract_id       VARCHAR(36)   PRIMARY KEY,
  listing_id        VARCHAR(36)   NOT NULL,

Axios Interceptor

import axios from 'axios';
import keycloak from './keycloak';

const apiClient = axios.create({
  baseURL: 'http://localhost:8080/api/v1',
});

apiClient.interceptors.request.use((config) => {
  if (keycloak.token) {
    config.headers.Authorization = `Bearer ${keycloak.token}`;
  }
  return config;
});

export default apiClient;

Pattern chung — áp dụng cho tất cả handlers

notification-service/.../EmailTemplates.java 
public class EmailTemplates {

    public static String contractSigned(String contractId, String productName,
                                              String agreedPrice, String deadline) {
        return """
            Hợp đồng đã được ký bởi cả hai bên.

            Mã hợp đồng  : %s
            Sản phẩm     : %s
            Giá thỏa thuận: %s VNĐ
            Hạn giao hàng : %s

            Vui lòng chờ xác nhận khóa escrow trước khi giao hàng.
            """.formatted(contractId, productName, agreedPrice, deadline);
    }

    public static String escrowLocked(String contractId, String amount, String buyerOrg) {
        return """
            Escrow đã được khóa. Bạn có thể tiến hành giao hàng.

            Mã hợp đồng : %s
            Số tiền lock : %s VNĐ
            Bên mua      : %s

            Sau khi bên mua xác nhận nhận hàng, thanh toán sẽ được giải phóng tự động.
            """.formatted(contractId, amount, buyerOrg);
    }

    public static String goodsDelivered(String contractId, String buyerOrg) {
        return """
            Bên mua đã xác nhận nhận hàng.

            Mã hợp đồng : %s
            Bên mua      : %s

            Hệ thống đang xử lý giải phóng thanh toán.
            """.formatted(contractId, buyerOrg);
    }

    public static String escrowReleased(String contractId, String amount, String settledAt) {
        return """
            Thanh toán đã được giải phóng thành công.

            Mã hợp đồng   : %s
            Số tiền nhận  : %s VNĐ
            Ngày thanh toán: %s

            Hợp đồng đã hoàn tất.
            """.formatted(contractId, amount, settledAt);
    }

    public static String contractCancelled(String contractId, String cancelledBy, String reason) {
        return """
            Hợp đồng đã bị hủy.

            Mã hợp đồng : %s
            Hủy bởi     : %s
            Lý do        : %s
            """.formatted(contractId, cancelledBy, reason);
    }

    public static String escrowPenalized(String contractId, String penalizedParty, String amount) {
        return """
            Thông báo: bạn bị phạt do vi phạm hợp đồng.

            Mã hợp đồng  : %s
            Bên bị phạt  : %s
            Số tiền phạt : %s VNĐ
            """.formatted(contractId, penalizedParty, amount);
    }

    public static String contractDisputed(String contractId, String buyerOrg, String timestamp) {
        return """
            [ADMIN] Có tranh chấp mới cần xử lý.

            Mã hợp đồng : %s
            Bên mua      : %s
            Thời gian    : %s

            Truy cập admin panel để phán xử.
            """.formatted(contractId, buyerOrg, timestamp);
    }
}

Sprint 3 · 07/07 – 20/07
🧪 Testing

Postman Collection — 3 Flows

File: docs/AgriContract-Phase1.postman_collection.json

Environment variables cần setup: baseUrl, sellerToken, buyerToken, adminToken, contractId, listingId, escrowId.

Flow 1 — Happy Path (SETTLED)

1
POST /api/v1/users/register (seller token)

Register seller. Save sellerId.

2
POST /api/v1/users/register (buyer token)

Register buyer. Save buyerId.

3
POST /api/v1/products (seller token)

Tạo product. Save productId.

4
POST /api/v1/listings (seller token)

Tạo listing. Save listingId. Assert status = ACTIVE.

5
POST /api/v1/contracts (buyer token)

Tạo contract từ listing. Save contractId. Assert status = OFFERED.

6
PUT /api/v1/contracts/{contractId}/sign (seller token)

Seller ký.

7
PUT /api/v1/contracts/{contractId}/sign (buyer token)

Buyer ký. Assert status = SIGNED. Listing → CLOSED.

8
GET /api/v1/escrows/contract/{contractId} — delay 2s

Đợi async saga. Assert status = LOCKED. Save escrowId.

9
PUT /api/v1/contracts/{contractId}/confirm-delivery (buyer token)

Assert status = GOODS_DELIVERED.

10
GET /api/v1/contracts/{contractId} — delay 2s

Assert status = SETTLED.

11
GET /api/v1/escrows/{escrowId}/transactions

Assert có đúng 2 transactions: LOCK + RELEASE.

12
MailHog check

localhost:8025 — phải có ≥3 emails (signed, escrow locked, settled).

Flow 2 — Buyer Cancel

Repeat bước 1–8, sau đó:

9
PUT /api/v1/contracts/{contractId}/cancel (buyer token)

Body: {"reason": "Không còn nhu cầu"}. Assert status = CANCELLED_BY_BUYER.

10
GET /api/v1/escrows/contract/{contractId} — delay 2s

Assert status = PENALIZED_BUYER. Transactions: LOCK + PENALIZE_BUYER + RELEASE.

Flow 3 — Dispute + Admin Arbitrate

Repeat bước 1–9 (đến GOODS_DELIVERED), sau đó:

10
PUT /api/v1/contracts/{contractId}/dispute (buyer token)

Body: {"reason": "Hàng không đúng chất lượng"}. Assert status = DISPUTED.

11
PUT /api/v1/escrows/{contractId}/arbitrate (admin token)

Body: {"buyerAmount": 30000000, "sellerAmount": 20000000}. Assert escrow status = ARBITRATED.

12
MailHog check

Email tranh chấp gửi admin + email kết quả cho cả hai bên.
